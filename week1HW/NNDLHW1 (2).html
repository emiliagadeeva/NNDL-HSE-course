# Visualize missing data
plt.figure(figsize=(12, 6))
missing = df.isnull().sum().sort_values(ascending=False)
missing_percent = (missing / len(df)) * 100
missing_df = pd.DataFrame({'Missing Count': missing, 'Percentage': missing_percent.round(2)})
missing_df = missing_df[missing_df['Missing Count'] > 0]

ax = sns.barplot(x=missing_df.index, y=missing_df['Percentage'], palette="Reds_r")
plt.title('🔍 Missing Values Percentage by Feature', fontsize=16, pad=20)
plt.ylabel('Percentage (%)', fontsize=12)
plt.xticks(rotation=45)
for i, v in enumerate(missing_df['Percentage']):
    ax.text(i, v + 1, f'{v}%', ha='center', fontweight='bold')
plt.tight_layout()
plt.show()

# Create survival visualization
survived_count = df['Survived'].value_counts()
labels = ['Perished', 'Survived']
colors = ['#ff9999','#66b3ff']

fig = make_subplots(rows=1, cols=2, specs=[[{'type':'domain'}, {'type':'xy'}]])

fig.add_trace(go.Pie(labels=labels, values=survived_count.values,
                     marker_colors=colors, hole=.3,
                     textinfo='label+percent'), 1, 1)

fig.add_trace(go.Bar(x=labels, y=survived_count.values,
                     marker_color=colors, text=survived_count.values,
                     textposition='auto'), 1, 2)

fig.update_layout(title_text='📈 Passenger Survival Distribution',
                  annotations=[dict(text='Survived?', x=0.18, y=0.5, font_size=15, showarrow=False)])
fig.show()

print(f"👥 Total Passengers: {len(df)}")

# Create comprehensive visualization
fig, axes = plt.subplots(2, 2, figsize=(16, 12))
fig.suptitle('🎭 Survival Analysis by Gender and Class', fontsize=20, y=0.95)

# 1. Survival by Gender
gender_survival = pd.crosstab(df['Sex'], df['Survived']).apply(lambda x: x/x.sum()*100, axis=1)
gender_survival.plot(kind='bar', ax=axes[0,0], color=['#ff6b6b', '#4ecdc4'])
axes[0,0].set_title('Survival Rate by Gender', fontsize=14)
axes[0,0].set_ylabel('Percentage (%)')
axes[0,0].legend(['Perished', 'Survived'])
axes[0,0].tick_params(axis='x', rotation=0)

# 2. Survival by Class
class_survival = pd.crosstab(df['Pclass'], df['Survived']).apply(lambda x: x/x.sum()*100, axis=1)
class_survival.plot(kind='bar', ax=axes[0,1], color=['#ff6b6b', '#4ecdc4'])
axes[0,1].set_title('Survival Rate by Passenger Class', fontsize=14)
axes[0,1].set_ylabel('Percentage (%)')
axes[0,1].legend(['Perished', 'Survived'])
axes[0,1].set_xticklabels(['1st Class', '2nd Class', '3rd Class'], rotation=0)

# 3. Passenger distribution by class and gender
sns.countplot(data=df, x='Pclass', hue='Sex', ax=axes[1,0], palette='coolwarm')
axes[1,0].set_title('Passenger Distribution by Class and Gender', fontsize=14)
axes[1,0].set_xlabel('Passenger Class')
axes[1,0].set_ylabel('Count')
axes[1,0].set_xticklabels(['1st Class', '2nd Class', '3rd Class'])
axes[1,0].legend(title='Gender')

axes[1,1].set_ylabel('Survival Probability')

for i, sex in enumerate(['male', 'female']):

        age_data = df[(df['Sex']==sex) & (df['Survived']==survived)]['Age'].dropna()
        fig.add_trace(go.Box(y=age_data, name=f"{'Male' if sex=='male' else 'Female'} - {'Perished' if survived==0 else 'Survived'}",

                     row=2, col=2)

# Create family features
